18def get_path(local_path):
19 """
20 Adds the last folder downloaded to root path
21 Args:
22 local_path (str): Root path where is saved files
23 Returns : The actual folder where files where downloaded
24 """
25 path_child = os.popen("ls {}".format(local_path)).read()
26 local_path_full = local_path+path_child
27 local_path_full = local_path_full.strip("\n")
28 local_path_renamed = local_path_full.replace(" ", "_")
29 os.rename(local_path_full, local_path_renamed)
30 if os.path.isdir(local_path_renamed):
31 return local_path_renamed+"/"
32 return local_path
