31def change_period(start_date):
32 """
33 Sum one month to period extracted from MongoDB
34 Args:
35 start_date (str): last period request (last period downloaded)
36 Returns : period (str)
37 """
38 y = int(start_date[:4])
39 m = int(start_date[4:])
40 start_date = date(y, m, 1)
41 year = start_date.year
42 month = start_date.month
43 if month == 12:
44 year += 1
45 month = 1
46 else:
47 month += 1
48
49 if month < 10:
50 month = "0"+str(month)
51 period = f"{year}{month}"
52 return period
